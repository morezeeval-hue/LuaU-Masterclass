Variablen, Typen, Funktionen

--!strict
local coins: number = 0
local playerName: string = "Builder"
local isAlive: boolean = true

local function addCoins(current: number, amount: number): number
    assert(amount >= 0, "amount must be positive")
    return current + amount
end

coins = addCoins(coins, 25)
print(playerName, coins, isAlive)

ModuleScript: InventoryService

--!strict
-- ServerScriptService/Services/InventoryService.lua
local InventoryService = {}

type Inventory = {[string]: number}
local inventories: {[Player]: Inventory} = {}

function InventoryService.InitPlayer(player: Player)
    inventories[player] = {}
end

function InventoryService.AddItem(player: Player, itemId: string, amount: number)
    if amount <= 0 then return false end
    local inv = inventories[player]
    if not inv then return false end
    inv[itemId] = (inv[itemId] or 0) + amount
    return true
end

function InventoryService.GetInventory(player: Player): Inventory?
    return inventories[player]
end

function InventoryService.RemovePlayer(player: Player)
    inventories[player] = nil
end

return InventoryService

RemoteEvent sicher validieren

--!strict
-- ReplicatedStorage/Remotes/BuyItem : RemoteEvent
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local BuyItem = ReplicatedStorage.Remotes.BuyItem

local PRICES = { sword = 100, potion = 25 }
local cooldown: {[Player]: number} = {}

BuyItem.OnServerEvent:Connect(function(player: Player, itemId: string)
    if typeof(itemId) ~= "string" then return end
    if not PRICES[itemId] then return end

    local now = os.clock()
    if cooldown[player] and now - cooldown[player] < 0.35 then return end
    cooldown[player] = now

    -- Server liest Geld selbst. Niemals Client-Werte vertrauen.
    local coins = player:GetAttribute("Coins") or 0
    local price = PRICES[itemId]
    if coins < price then return end

    player:SetAttribute("Coins", coins - price)
    -- InventoryService.AddItem(player, itemId, 1)
end)

DataStore-Grundstruktur mit pcall

--!strict
local DataStoreService = game:GetService("DataStoreService")
local Players = game:GetService("Players")
local store = DataStoreService:GetDataStore("PlayerData_v1")

type SaveData = { coins: number, level: number, inventory: {[string]: number} }
local session: {[Player]: SaveData} = {}

local function defaultData(): SaveData
    return { coins = 0, level = 1, inventory = {} }
end

local function loadPlayer(player: Player)
    local ok, data = pcall(function()
        return store:GetAsync(tostring(player.UserId))
    end)
    if ok and typeof(data) == "table" then
        session[player] = data :: SaveData
    else
        session[player] = defaultData()
    end
end

local function savePlayer(player: Player)
    local data = session[player]
    if not data then return end
    local ok, err = pcall(function()
        store:SetAsync(tostring(player.UserId), data)
    end)
    if not ok then warn("Save failed", player, err) end
end

Players.PlayerAdded:Connect(loadPlayer)
Players.PlayerRemoving:Connect(function(player)
    savePlayer(player)
    session[player] = nil
end)

CollectionService Component Pattern

--!strict
local CollectionService = game:GetService("CollectionService")

local function setupDamagePart(part: Instance)
    if not part:IsA("BasePart") then return end
    part.Touched:Connect(function(hit)
        local humanoid = hit.Parent and hit.Parent:FindFirstChildOfClass("Humanoid")
        if humanoid then humanoid:TakeDamage(10) end
    end)
end

for _, inst in ipairs(CollectionService:GetTagged("DamagePart")) do
    setupDamagePart(inst)
end
CollectionService:GetInstanceAddedSignal("DamagePart"):Connect(setupDamagePart)

PathfindingService NPC

--!strict
local PathfindingService = game:GetService("PathfindingService")

local npc = script.Parent
local humanoid = npc:WaitForChild("Humanoid") :: Humanoid
local root = npc:WaitForChild("HumanoidRootPart") :: BasePart

local function moveTo(goal: Vector3)
    local path = PathfindingService:CreatePath()
    path:ComputeAsync(root.Position, goal)
    if path.Status ~= Enum.PathStatus.Success then return end
    for _, waypoint in ipairs(path:GetWaypoints()) do
        humanoid:MoveTo(waypoint.Position)
        humanoid.MoveToFinished:Wait()
    end
end

moveTo(Vector3.new(0, 0, 50))

MarketplaceService Developer Product

--!strict
local MarketplaceService = game:GetService("MarketplaceService")

local PRODUCT_REWARDS = {
    [123456789] = function(player: Player)
        player:SetAttribute("Coins", (player:GetAttribute("Coins") or 0) + 1000)
    end,
}

MarketplaceService.ProcessReceipt = function(receiptInfo)
    local player = game.Players:GetPlayerByUserId(receiptInfo.PlayerId)
    if not player then
        return Enum.ProductPurchaseDecision.NotProcessedYet
    end
    local reward = PRODUCT_REWARDS[receiptInfo.ProductId]
    if reward then reward(player) end
    return Enum.ProductPurchaseDecision.PurchaseGranted
end

Was du vor einem schwierigen Projekt können solltest

  • Du weißt, welcher Code auf dem Server und welcher auf dem Client läuft.
  • Du vertraust niemals Client-Daten bei Geld, Damage, Items oder Progress.
  • Du kannst ModuleScripts als klare APIs schreiben.
  • Du kannst RemoteEvents mit Typprüfung, Cooldown und Validierung absichern.
  • Du kannst Daten mit Fehlerbehandlung speichern und laden.
  • Du kannst UI-State vom Gameplay-State trennen.
  • Du kannst Performance-Probleme messen statt raten.
